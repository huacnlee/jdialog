jQuery jDialog plugin

About

This is a jQuery UI plugin,You can use simple method to create a popup window.

Source code

How to use

            
function changeLocal(el){
  $(el).jDialog({
    title : "Quick choice",
    content : 'This is dialog content. ',
    width : 200
  });
  return false;
}

function showSearchPanel(el){
  $(el).jDialog({
    title : "Quick choice",
    content : $("#hide_contexts .search_box").clone().html(),
    width : 320
  });
  return false;
}

function showOnlyCloseButtonPanel(el){
  $(el).jDialog({
    title : "Close only with button",
    content : 'This window only close with the Clonse button.',
    width : 320,
    close_on_body_click : false
  });
  return false;
}
